```python
"""实验 G 工具库 —— Ordinal Margin Ranking Loss。

设计目标
  baseline 在 score 空间下 benign / no_tumor 高度重叠。在 BCE 之上加 pairwise
  margin loss,显式约束:
      sigmoid(score_mal)  >= sigmoid(score_ben) + margin
      sigmoid(score_ben)  >= sigmoid(score_nt)  + margin

  margin 操作的是 post-sigmoid score (in [0,1]),不是 logit。
  这样 ben 不再混入 low band,nt 不再混入 high band。

只新增 loss,不新增模型类 —— 复用 baseline 的 SwinV2SegGuidedRiskTrimodal。
"""
```

```python
class RiskOrdinalMarginLoss(nn.Module):
    """L = lambda_ord * BCE_logits(risk_logit, ord_target)
         + lambda_rank * margin_ranking_loss(sigmoid(logit), label)
         + (CE + Dice)(seg)
    """
    def __init__(self, ordinal_targets: torch.Tensor,
                 lambda_ord: float = 2.0,
                 lambda_rank: float = 0.3,
                 margin: float = 0.2,
                 seg_ce_weight: Optional[torch.Tensor] = None):
        super().__init__()
        self.register_buffer("targets", ordinal_targets)
        self.lambda_ord = lambda_ord
        self.lambda_rank = lambda_rank
        self.margin = margin
        self.seg_ce = nn.CrossEntropyLoss(weight=seg_ce_weight)
        self.seg_dice = DiceLoss()

    def forward(self, seg_logits, risk_logit, seg_targets, cls_targets, has_mask):
        # ── BCE ord loss ──
        ord_target = self.targets[cls_targets].to(risk_logit.dtype)
        ord_loss = F.binary_cross_entropy_with_logits(risk_logit, ord_target)

        # ── Margin ranking loss (post-sigmoid space) ──
        # 注意 AMP 安全: 在 fp32 算 sigmoid 和 ranking,避免数值不稳
        score_fp32 = torch.sigmoid(risk_logit.float())  # (B,) in [0, 1]
        rank_loss = self._margin_ranking(score_fp32, cls_targets)

        # ── Seg loss (仅在 has_mask 上) ──
        seg_loss = torch.tensor(0.0, device=risk_logit.device)
        if has_mask.any():
            idx = has_mask.nonzero(as_tuple=True)[0]
            seg_ce = self.seg_ce(seg_logits[idx], seg_targets[idx])
            seg_dice = self.seg_dice(seg_logits[idx], seg_targets[idx])
            seg_loss = seg_ce + seg_dice

        total = (seg_loss + self.lambda_ord * ord_loss
                 + self.lambda_rank * rank_loss.to(risk_logit.dtype))
        return total, seg_loss.item(), ord_loss.item(), float(rank_loss.item())

    def _margin_ranking(self, score: torch.Tensor, label: torch.Tensor) -> torch.Tensor:
        """label: 0=mal, 1=ben, 2=nt (与 ORDINAL_TARGETS_DEFAULT 顺序一致)
        要求: score(mal) > score(ben) + margin AND score(ben) > score(nt) + margin
        """
        s_mal = score[label == 0]
        s_ben = score[label == 1]
        s_nt  = score[label == 2]
        loss = score.new_zeros(())
        # (mal, ben) 对: 期望 s_mal - s_ben > margin
        if s_mal.numel() > 0 and s_ben.numel() > 0:
            diff = s_mal.unsqueeze(1) - s_ben.unsqueeze(0)  # (n_mal, n_ben)
            loss = loss + F.relu(self.margin - diff).mean()
        # (ben, nt) 对: 期望 s_ben - s_nt > margin
        if s_ben.numel() > 0 and s_nt.numel() > 0:
            diff = s_ben.unsqueeze(1) - s_nt.unsqueeze(0)
            loss = loss + F.relu(self.margin - diff).mean()
        return loss
```

```python
import numpy as np
from typing import Dict, Tuple

def search_thresholds_g(scores: np.ndarray, labels: np.ndarray,
                        t_low_range: Tuple[float, float, float] = (0.05, 0.45, 0.02),
                        t_high_range: Tuple[float, float, float] = (0.50, 0.95, 0.02),
                        max_benign_to_low: float = 0.05,
                        max_nt_to_high: float = 0.05) -> Dict:
    """G 的阈值搜索: 沿用 baseline 双阈值 + 新软底线。

    硬底线: mal_to_low = 0, high_recall >= 0.95
    新软底线: benign_to_low <= 5%, no_tumor_to_high <= 5%
    目标: maximize (high_precision + low_precision_nt) / 2
    若无解: 软底线放宽到 benign_to_low <= 10%, no_tumor_to_high <= 10% 重搜。
    """
    def _eval(t_low, t_high):
        bands = np.where(scores >= t_high, 0,
                         np.where(scores <= t_low, 2, 1))
        n_mal = int((labels == 0).sum())
        n_ben = int((labels == 1).sum())
        n_nt  = int((labels == 2).sum())
        mal_to_low = int(((labels == 0) & (bands == 2)).sum())
        ben_to_low = int(((labels == 1) & (bands == 2)).sum())
        nt_to_high = int(((labels == 2) & (bands == 0)).sum())
        high_recall = ((labels == 0) & (bands == 0)).sum() / max(n_mal, 1)
        n_high = int((bands == 0).sum())
        n_low = int((bands == 2).sum())
        high_precision = (((labels == 0) & (bands == 0)).sum() /
                          n_high) if n_high > 0 else 0.0
        low_precision_nt = (((labels == 2) & (bands == 2)).sum() /
                            n_low) if n_low > 0 else 0.0
        return dict(
            t_low=t_low, t_high=t_high, mal_to_low=mal_to_low,
            high_recall=float(high_recall),
            high_precision=float(high_precision),
            low_precision_nt=float(low_precision_nt),
            ben_to_low_share=ben_to_low / max(n_ben, 1),
            nt_to_high_share=nt_to_high / max(n_nt, 1),
        )

    def _search(max_ben_low, max_nt_high):
        best = None
        for t_low in np.arange(*t_low_range):
            for t_high in np.arange(*t_high_range):
                if t_high - t_low < 0.05:
                    continue
                m = _eval(float(t_low), float(t_high))
                if m["mal_to_low"] != 0:                 continue
                if m["high_recall"] < 0.95:              continue
                if m["ben_to_low_share"] > max_ben_low:  continue
                if m["nt_to_high_share"] > max_nt_high:  continue
                obj = (m["high_precision"] + m["low_precision_nt"]) / 2
                m["objective"] = obj
                if best is None or obj > best["objective"]:
                    best = m
        return best

    best = _search(max_benign_to_low, max_nt_to_high)
    if best is None:
        best = _search(0.10, 0.10)  # fallback
        if best is not None:
            best["fallback_used"] = True
    if best is None:
        # 极端 fallback: 退化到 baseline 双阈值搜索口径(只看硬底线)
        best = _search(1.0, 1.0)
        best["fallback_used"] = "no_soft_floor"
    return best
```

```python
def forward(self, x, metadata=None, input_ids=None, attention_mask=None):
    features = self.encoder(x)
    f0, f1, f2, f3 = [self._to_bchw(f) for f in features]
    d3 = self.dec3(f3, f2); d2 = self.dec2(d3, f1); d1 = self.dec1(d2, f0)
    seg_logits = self.seg_final(d1)
    f2_proj = self.cls_proj(f2)
    with torch.no_grad():
        text_out = self.text_encoder(input_ids=input_ids, attention_mask=attention_mask)
    text_hidden = text_out.last_hidden_state
    text_cls = text_hidden[:, 0]
    f2_enhanced = self.cross_attn(f2_proj, text_hidden, attention_mask)
    seg_prob = F.softmax(seg_logits, dim=1)[:, 1:2]
    attn = F.interpolate(seg_prob, size=f2_enhanced.shape[2:],
                         mode="bilinear", align_corners=False)
    attn = attn + 0.1
    attn = attn / (attn.sum(dim=(2, 3), keepdim=True) + 1e-6)
    img_feat = (f2_enhanced * attn).sum(dim=(2, 3))
    text_feat = self.text_cls_proj(text_cls)
    meta_feat = self.meta_encoder(metadata.float())
    fused = self.fusion(img_feat, text_feat, meta_feat)
    risk_logit = self.risk_head(fused).squeeze(-1)
    cls_logits = self.cls_head(fused)  # ← E 新增
    return seg_logits, risk_logit, cls_logits  # ← 返回多一个
```

```python
class RiskOrdinalCEL(nn.Module):
    """L = lambda_ord * BCE_logits(risk, ord) + lambda_cls * CE(cls_logits, label)
         + (CE + Dice)(seg)
    label 顺序: 0=mal, 1=ben, 2=nt (与 cls_head output 对齐)
    """
    def __init__(self, ordinal_targets, lambda_ord=2.0, lambda_cls=0.5,
                 seg_ce_weight=None):
        super().__init__()
        self.register_buffer("targets", ordinal_targets)
        self.lambda_ord = lambda_ord
        self.lambda_cls = lambda_cls
        self.cls_ce = nn.CrossEntropyLoss()
        self.seg_ce = nn.CrossEntropyLoss(weight=seg_ce_weight)
        self.seg_dice = DiceLoss()

    def forward(self, seg_logits, risk_logit, cls_logits,
                seg_targets, cls_targets, has_mask):
        ord_target = self.targets[cls_targets].to(risk_logit.dtype)
        ord_loss = F.binary_cross_entropy_with_logits(risk_logit, ord_target)
        cls_loss = self.cls_ce(cls_logits, cls_targets)
        seg_loss = torch.tensor(0.0, device=risk_logit.device)
        if has_mask.any():
            idx = has_mask.nonzero(as_tuple=True)[0]
            seg_loss = (self.seg_ce(seg_logits[idx], seg_targets[idx]) +
                        self.seg_dice(seg_logits[idx], seg_targets[idx]))
        total = (seg_loss + self.lambda_ord * ord_loss
                 + self.lambda_cls * cls_loss)
        return total, seg_loss.item(), ord_loss.item(), cls_loss.item()
```

```python
def decide_band_aux_cls(risk_score, p_nt, p_mal, p_ben,
                        t_low, t_high, tau_nt, tau_mal, tau_ben):
    """单样本决策。返回 0=high, 1=med, 2=low。"""
    if p_nt > tau_nt:    return 2  # low
    if p_mal > tau_mal:  return 0  # high
    if p_ben > tau_ben:  return 1  # medium
    if risk_score >= t_high: return 0
    if risk_score <= t_low:  return 2
    return 1


def decide_bands_aux_cls_vec(risk_scores, cls_probs, t_low, t_high,
                              tau_nt, tau_mal, tau_ben):
    """向量化版本。cls_probs: (N, 3) softmax 后,顺序 [mal, ben, nt]"""
    p_mal, p_ben, p_nt = cls_probs[:, 0], cls_probs[:, 1], cls_probs[:, 2]
    bands = np.full(len(risk_scores), 1, dtype=np.int64)  # 默认 medium
    bands = np.where(p_nt > tau_nt, 2, bands)              # nt 优先 low
    bands = np.where(p_mal > tau_mal, 0, bands)             # mal 优先 high (覆盖前一步)
    # ben 优先级低于 mal/nt: 仅当前两步未触发时才 medium
    not_routed = (p_nt <= tau_nt) & (p_mal <= tau_mal)
    bands = np.where(not_routed & (p_ben > tau_ben), 1, bands)
    # 否则用 risk 双阈值
    not_routed_all = not_routed & (p_ben <= tau_ben)
    bands = np.where(not_routed_all & (risk_scores >= t_high), 0, bands)
    bands = np.where(not_routed_all & (risk_scores <= t_low), 2, bands)
    return bands
```

> ⚠ 注意决策优先级实现。spec §3.4 的伪代码是 if/elif/elif,即 nt > mal > ben > risk。
> 上面的向量化实现等效:先填默认 medium,再按反向覆盖。

```python
def search_thresholds_e(risk_scores, cls_probs, labels,
                        t_low_range=(0.05, 0.45, 0.05),
                        t_high_range=(0.50, 0.95, 0.05),
                        tau_range=(0.30, 0.71, 0.05),  # tau_nt, tau_mal, tau_ben 共享范围
                        max_ben_to_low=0.05,
                        max_nt_to_high=0.05):
    """5D grid: (t_low, t_high, tau_nt, tau_mal, tau_ben)
    实际节点数 9 * 10 * 9 * 9 * 9 = 65610,在 val 集上跑一秒级。
    """
    best = None
    for t_low in np.arange(*t_low_range):
        for t_high in np.arange(*t_high_range):
            if t_high - t_low < 0.05: continue
            for tau_nt in np.arange(*tau_range):
                for tau_mal in np.arange(*tau_range):
                    for tau_ben in np.arange(*tau_range):
                        bands = decide_bands_aux_cls_vec(
                            risk_scores, cls_probs,
                            float(t_low), float(t_high),
                            float(tau_nt), float(tau_mal), float(tau_ben))
                        m = _band_metrics(bands, labels)
                        if not _passes_constraints(m, max_ben_to_low, max_nt_to_high):
                            continue
                        obj = (m["high_precision"] + m["low_precision_nt"]) / 2
                        if best is None or obj > best["objective"]:
                            best = {**m, "t_low": float(t_low), "t_high": float(t_high),
                                    "tau_nt": float(tau_nt), "tau_mal": float(tau_mal),
                                    "tau_ben": float(tau_ben), "objective": obj}
    if best is None:
        # fallback: 把软底线从 0.05 放宽到 0.10,重搜
        return search_thresholds_e(
            risk_scores, cls_probs, labels,
            t_low_range=t_low_range, t_high_range=t_high_range,
            tau_range=tau_range,
            max_ben_to_low=0.10, max_nt_to_high=0.10) if max_ben_to_low < 0.10 \
            else _fallback_no_soft(risk_scores, cls_probs, labels)
    best["fallback_used"] = max_ben_to_low > 0.05
    return best


def _fallback_no_soft(risk_scores, cls_probs, labels):
    """极端 fallback: 完全去掉软底线,只看硬底线 (mal_to_low=0, high_recall>=0.95)。"""
    # 退化为只用 risk_score 双阈值,tau_* 都设为 1.1 (永远不触发类别先验)
    from risk_utils import search_risk_thresholds_constrained, PRIMARY_PROFILE
    res = search_risk_thresholds_constrained(risk_scores, labels, profile=PRIMARY_PROFILE)
    return {**res, "tau_nt": 1.1, "tau_mal": 1.1, "tau_ben": 1.1,
            "fallback_used": "no_soft_floor"}


def _band_metrics(bands, labels):
    n_mal = int((labels == 0).sum())
    n_ben = int((labels == 1).sum())
    n_nt  = int((labels == 2).sum())
    mal_to_low  = int(((labels == 0) & (bands == 2)).sum())
    high_recall = ((labels == 0) & (bands == 0)).sum() / max(n_mal, 1)
    n_high = int((bands == 0).sum())
    n_low  = int((bands == 2).sum())
    return dict(
        mal_to_low=mal_to_low,
        high_recall=float(high_recall),
        high_precision=float(((labels==0) & (bands==0)).sum() / n_high) if n_high else 0.0,
        low_precision_nt=float(((labels==2) & (bands==2)).sum() / n_low) if n_low else 0.0,
        ben_to_low_share=float(((labels==1) & (bands==2)).sum() / max(n_ben, 1)),
        nt_to_high_share=float(((labels==2) & (bands==0)).sum() / max(n_nt, 1)),
    )


def _passes_constraints(m, max_ben_low, max_nt_high):
    if m["mal_to_low"] != 0:                  return False
    if m["high_recall"] < 0.95:               return False
    if m["ben_to_low_share"] > max_ben_low:   return False
    if m["nt_to_high_share"] > max_nt_high:   return False
    return True
```

```python
class TripleBCELoss(nn.Module):
    """L = BCE(p_high, y_high) + BCE(p_med, y_med) + BCE(p_low, y_low)
         + (CE + Dice)(seg)
    label 顺序: 0=mal, 1=ben, 2=nt
    y_high = (label==0); y_med = (label==1); y_low = (label==2)
    """
    def __init__(self, seg_ce_weight=None):
        super().__init__()
        self.seg_ce = nn.CrossEntropyLoss(weight=seg_ce_weight)
        self.seg_dice = DiceLoss()

    def forward(self, seg_logits, logit_high, logit_med, logit_low,
                seg_targets, cls_targets, has_mask):
        y_high = (cls_targets == 0).to(logit_high.dtype)
        y_med  = (cls_targets == 1).to(logit_high.dtype)
        y_low  = (cls_targets == 2).to(logit_high.dtype)
        bce_high = F.binary_cross_entropy_with_logits(logit_high, y_high)
        bce_med  = F.binary_cross_entropy_with_logits(logit_med,  y_med)
        bce_low  = F.binary_cross_entropy_with_logits(logit_low,  y_low)
        seg_loss = torch.tensor(0.0, device=logit_high.device)
        if has_mask.any():
            idx = has_mask.nonzero(as_tuple=True)[0]
            seg_loss = (self.seg_ce(seg_logits[idx], seg_targets[idx]) +
                        self.seg_dice(seg_logits[idx], seg_targets[idx]))
        total = seg_loss + bce_high + bce_med + bce_low
        return total, seg_loss.item(), float(bce_high), float(bce_med), float(bce_low)
```

```python
class CascadeLoss(nn.Module):
    """L = BCE(p_tumor, y_tumor) + lambda_mal * BCE_weighted(p_mal, y_mal | tumor)
         + (CE + Dice)(seg)
    label 顺序: 0=mal, 1=ben, 2=nt
    y_tumor = (label != 2);  y_mal = (label == 0)
    mal head 的 loss 用 y_tumor.float() 加权 (no_tumor 样本梯度=0)
    """
    def __init__(self, lambda_mal=1.0, seg_ce_weight=None):
        super().__init__()
        self.lambda_mal = lambda_mal
        self.seg_ce = nn.CrossEntropyLoss(weight=seg_ce_weight)
        self.seg_dice = DiceLoss()

    def forward(self, seg_logits, tumor_logit, mal_logit,
                seg_targets, cls_targets, has_mask):
        y_tumor = (cls_targets != 2).to(tumor_logit.dtype)
        y_mal   = (cls_targets == 0).to(tumor_logit.dtype)
        loss_tumor = F.binary_cross_entropy_with_logits(tumor_logit, y_tumor)
        # mal head: 只在 tumor 样本上有梯度
        mal_loss_per = F.binary_cross_entropy_with_logits(
            mal_logit, y_mal, reduction='none')
        loss_mal = ((mal_loss_per * y_tumor).sum() /
                    (y_tumor.sum() + 1e-6))
        seg_loss = torch.tensor(0.0, device=tumor_logit.device)
        if has_mask.any():
            idx = has_mask.nonzero(as_tuple=True)[0]
            seg_loss = (self.seg_ce(seg_logits[idx], seg_targets[idx]) +
                        self.seg_dice(seg_logits[idx], seg_targets[idx]))
        total = seg_loss + loss_tumor + self.lambda_mal * loss_mal
        return total, seg_loss.item(), float(loss_tumor), float(loss_mal)
```

- [ ] **Step 4: 写决策规则**

```python
def decide_bands_cascade(p_tumor, p_mal, tau1, tau2):
    """tau1: tumor threshold; tau2: mal threshold (in tumor pool)
    p_tumor < tau1   → low (no_tumor)
    p_mal   > tau2   → high (mal)
    else             → medium (ben)
    """
    bands = np.full(len(p_tumor), 1, dtype=np.int64)  # 默认 medium
    bands = np.where(p_tumor < tau1, 2, bands)         # 低 tumor → low
    bands = np.where((p_tumor >= tau1) & (p_mal > tau2), 0, bands)
    return bands
```

- [ ] **Step 5: 写两阶段阈值搜索**

```python
def search_thresholds_cascade(p_tumor, p_mal, labels,
                              tau1_range=(0.10, 0.71, 0.02),
                              tau2_range=(0.30, 0.91, 0.02),
                              max_ben_to_low=0.05,
                              max_nt_to_high=0.05):
    """先按 tumor_recall >= 0.98 缩 tau1,再 grid search tau2 + 软底线。"""
    y_tumor = (labels != 2).astype(np.int64)
    n_tumor = int(y_tumor.sum())
    best = None
    for tau1 in np.arange(*tau1_range):
        # 硬底线 1: tumor recall >= 0.98 (不漏 tumor)
        tumor_recall = ((y_tumor == 1) & (p_tumor >= tau1)).sum() / max(n_tumor, 1)
        if tumor_recall < 0.98:  continue
        for tau2 in np.arange(*tau2_range):
            bands = decide_bands_cascade(p_tumor, p_mal, float(tau1), float(tau2))
            m = _band_metrics(bands, labels)  # 复用 E 同名函数
            if not _passes_constraints(m, max_ben_to_low, max_nt_to_high):
                continue
            obj = (m["high_precision"] + m["low_precision_nt"]) / 2
            if best is None or obj > best["objective"]:
                best = {**m, "tau1": float(tau1), "tau2": float(tau2),
                        "tumor_recall": float(tumor_recall), "objective": obj}
    if best is None:
        # fallback: 软底线放宽到 0.10/0.10 重搜一次
        if max_ben_to_low < 0.10:
            return search_thresholds_cascade(
                p_tumor, p_mal, labels,
                tau1_range=tau1_range, tau2_range=tau2_range,
                max_ben_to_low=0.10, max_nt_to_high=0.10)
        # 极端 fallback: 不要软底线,只看硬底线
        return _cascade_fallback_no_soft(p_tumor, p_mal, labels,
                                          tau1_range, tau2_range)
    best["fallback_used"] = max_ben_to_low > 0.05
    return best


def _cascade_fallback_no_soft(p_tumor, p_mal, labels, tau1_range, tau2_range):
    """硬底线 only (tumor_recall>=0.98 + mal_to_low=0 + high_recall>=0.95)。"""
    y_tumor = (labels != 2).astype(np.int64)
    n_mal = int((labels == 0).sum())
    n_tumor = int(y_tumor.sum())
    best = None
    for tau1 in np.arange(*tau1_range):
        if ((y_tumor == 1) & (p_tumor >= tau1)).sum() / max(n_tumor, 1) < 0.98:
            continue
        for tau2 in np.arange(*tau2_range):
            bands = decide_bands_cascade(p_tumor, p_mal, float(tau1), float(tau2))
            mal_to_low = ((labels == 0) & (bands == 2)).sum()
            high_recall = ((labels == 0) & (bands == 0)).sum() / max(n_mal, 1)
            if mal_to_low != 0:    continue
            if high_recall < 0.95: continue
            n_high = int((bands == 0).sum())
            high_p = ((labels == 0) & (bands == 0)).sum() / n_high if n_high else 0
            if best is None or high_p > best["high_precision"]:
                best = dict(tau1=float(tau1), tau2=float(tau2),
                            high_precision=float(high_p), high_recall=float(high_recall),
                            mal_to_low=int(mal_to_low),
                            fallback_used="no_soft_floor")
    return best
```
